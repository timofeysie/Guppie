/**
* @fileOverview The main controller takes care of actions for
* the Material Design app actions such as left and right panel
* navigation and the shout outs list.
* @author Timothy Curchod
* @version: 0.1
* @memberOf main

angular.module('gulpAngular')
  .controller('MainCtrl', ['$scope', 
    '$mdSidenav', 
    '$timeout',
    '$log',
    'userInfo',
    function ($scope, $mdSidenav, $timeout, $log, userInfo) {
    $scope.role = 'visitor';
    userInfo.setRole($scope.role);
    //var vm = this;
    $scope.updateRole = function () {
      //console.log('Role: '+$scope.role);
      userInfo.setRole($scope.role);
    };

    $scope.getRole = function () {
      //console.log('role: '+userInfo.getRole());
      return userInfo.getRole();
    }

    // material
    $scope.toggleLeft = buildToggler('left');
    $scope.toggleRight = buildToggler('right');
    /**
     * Build handler to open/close a SideNav; when animation finishes
     * report completion in console
     */
    function buildToggler(navID) {
      return function() {
        return $mdSidenav(navID).toggle()
          .then(function () {
            $log.debug("toggle " + navID + " is done");
          });
      }
    }

    $scope.awesomeThings = [
